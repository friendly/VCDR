data(Arthritis)

# define Better
Arthritis$Better <- Arthritis$Improved > "None"

# simple linear regression
arth.mod0 <- glm(Better ~ Age, data = Arthritis, family = "binomial")
anova(arth.mod0)

pred <- predict(arth.mod0, type = "response", se.fit = TRUE)

# smoothed non-parametric curve

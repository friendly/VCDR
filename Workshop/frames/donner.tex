\begin{frame}[fragile]
\frametitle{Quadratic model?}
\begin{itemize}
	\item Fit: $\Pr(\mbox{Death}) \sim \mbox{Age} + \mbox{Age}^2 + \mbox{Male}$
	\item Statistical evidence for Age$^2$ equivocal:
	\begin{itemize*} 
		\item Wald $\chisq_{(1)} = 2.84, p = 0.09$; but
		\item LR $\GSQ_{(1)} = 4.40, p= 0.03$. ...
	\end{itemize*}
\end{itemize}

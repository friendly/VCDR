# plots on the logit scale, from Denis Murphy

mod <- glm(survived ~ age * sex, family = binomial, data = Titanicp)
modp <- cbind(Titanicp[, c("survived", "sex", "age")], predict(mod, se.fit = TRUE))
modp$obs <- c(-2.5, 2.5)[modp$survived]

# Plot predicted logits with corresponding Wald CIs
p <- ggplot(modp, aes(x = age, y = fit, color = sex)) + geom_line(size = 2) + 
    geom_ribbon(aes(ymin = fit - 1.96 * se.fit, ymax = fit + 1.96 * se.fit, 
        fill = sex), alpha = 0.2, color = "transparent") + labs(x = "Age", y = "Estimated logit")
# p + geom_point(y=modp$obs, position=position_jitter(y=modp$obs,

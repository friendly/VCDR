\begin{listing}
data employ;
   input length $ @;
   do cause ='Close  ', 'Replace';
      do employed = 'Yes', 'No';
         input count @;
         output;
         end;
      end;
   input;
datalines;

*-- Reshape as two column matrix ([CL][E]);
proc sort data=employ;
   by cause length employed ;
proc transpose prefix=e out=employ2;
   var count;
   by cause length;
data employ2;
   set employ2;
   drop _name_;
   id = substr(cause,1,1) || length;

axis1 order=(-2 to 1) length=4.875in
   label=(a=90);

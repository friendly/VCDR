\begin{listing}
%let N=12;
%let lambda = 2, 5;
title "Poisson distributions, lambda=&lambda, k=0..&N";
data poisson;
   reps = 1000;
   drop reps;
   N=&N;
   do lambda=&lambda;
      do k=0 to N;
         prob = pdf('poisson', k, lambda);
         freq = reps * prob;
         output;
         end;
      end;
   label freq='Frequency'
        lambda='Lambda'
        k = 'k';
\end{listing}

\newcommand{\newtridot}[1]{%
	\begin{pspicture}(-.01, -.01)(1.0,1.0)
	\psset{xunit=.85,yunit=.85}
	\color{black}
	\rput(0,0){\circlenode{A}{\textsf{A}}}
	\rput(1.0,0){\circlenode{B}{\textsf{B}}}
	\rput(.5,.866){\circlenode{C}{\textsf{C}}}
	#1%
	\end{pspicture}
	\rule{0in}{1.1cm}
}
